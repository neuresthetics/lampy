def letterBlock(phrase, offset, interval):

  sentence = phrase
  words = sentence.split()
  output = ''
  lettercount = 0
  returnString = ''

  for word in words:
    for letter in word:
      lettercount += 1
      output += (chr(ord(letter)+offset))
    output += ' '

  returnString += output + "\n"
  sentence = output
  words = sentence.split()
  output = ''

  for k in range(51):
    for word in words:
      for letter in word:
        output += (chr(ord(letter)+interval))
      output += ' '

    returnString += output + "\n"
    sentence = output
    words = sentence.split()
    output = ''
  
  return returnString
